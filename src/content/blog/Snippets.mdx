`format-and-lint.yml`

```yml
name: Format and Lint

on:
  pull_request:
    branches:
      - main

jobs:
  format-and-lint:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout Repository
        uses: actions/checkout@v3

      - name: Node lts
        uses: actions/setup-node@v3
        with:
          node-version: 'latest'

      - name: Install Dependencies
        run: npm install

      - name: Format Code
        run: npm run format

      - name: Lint Code
        run: npm run lint
```
